-- Collection: Splunk - Missing Agent
-- Query rule name: Missing Splunk Universal Forwarder

select
    SMS_R_System.ResourceId,
    SMS_R_System.Name,
    SMS_R_System.OperatingSystemNameandVersion
from SMS_R_System
where SMS_R_System.ResourceId not in (
    select SMS_G_System_ADD_REMOVE_PROGRAMS.ResourceId
    from SMS_G_System_ADD_REMOVE_PROGRAMS
    where SMS_G_System_ADD_REMOVE_PROGRAMS.DisplayName like '%Splunk Universal Forwarder%'
)
